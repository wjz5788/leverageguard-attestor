  localStorage.setItem("lang", lang);

function renderI18n() {
  const lang = localStorage.getItem("lang") || "en";
  document.querySelectorAll("[data-i18n]").forEach(el=>{
    const key = el.getAttribute("data-i18n");
    let vars = {};
    try { vars = JSON.parse(el.getAttribute("data-i18n-vars")||"{}"); } catch{}
    const html = t(key, vars, lang);
    // 如果内容含有 <b> 等标签就用 innerHTML，否则用 textContent
    if (/[<>&]/.test(html)) el.innerHTML = html; else el.textContent = html;
  });
}

  document.querySelectorAll('.lang-switcher button').forEach(btn=>{

  });
}
document.querySelectorAll('.lang-switcher button').forEach(btn=>{
  btn.addEventListener('click', ()=> setLang(btn.dataset.lang));
});

// 初始化
if(!localStorage.getItem("lang")) localStorage.setItem("lang", getLang());
setLang(localStorage.getItem("lang"));

## ③ 给页面文字打上 `data-i18n`（示例）

把你现有 DOM 的文本换成带 key 的占位，变量（如交易所名）用 `data-i18n-vars` 传入：

```html
<!-- 顶部徽标/标题 -->
<p class="text-xs opacity-70" data-i18n="hero_badge"></p>

<h1 data-i18n="hero_flow" data-i18n-vars='{"exchange":"OKX"}'></h1>

<h2 data-i18n="products_title"></h2>

<h2 data-i18n="bind_title" data-i18n-vars='{"exchange":"OKX"}'></h2>

<!-- UX 步骤卡片示例 -->
<h3 data-i18n="step1_title"></h3>
<p  data-i18n="step1_desc"></p>

<h3 data-i18n="step2_title"></h3>
<p  data-i18n="step2_desc"></p>

<!-- 为什么在 Base -->
<h3 data-i18n="why_title"></h3>
<ul>
  <li data-i18n="why_a"></li>
  <li data-i18n="why_b"></li>
  <li data-i18n="why_c"></li>
</ul>
```

> 想把“OKX”换成当前用户选择的交易所时，只要用脚本更新该区域的 `data-i18n-vars='{"exchange":"Binance"}'` 再调用一次 `renderI18n()` 就行。
